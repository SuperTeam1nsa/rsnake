91/// Display the fruit menu, vertically centered
92fn fruit_menu(frame: &mut Frame, selected: &GreetingOption) {
93    //adding fruits rules, gonna left aligned for less screen space use
94    let mut fruits_lines = Vec::new();
95    let tab_jonction = Line::from("+---------+---------------------+--------+-------------+");
96    fruits_lines.push(tab_jonction.clone());
97    fruits_lines.push(Line::from(
98        "| Fruit   | Base Score (xSpeed) | Chance | Size Effect |",
99    ));
100    fruits_lines.push(tab_jonction.clone());
101    for (fruit, score, probability, size_effect) in FRUITS_SCORES_PROBABILITIES {
102        //:<6 and so on are formating options, e.g., saying aligning left with min 6 chars
103        fruits_lines.push(Line::from(format!(
104            "| {fruit:<6} | {score:>19} | {probability:>5}% | {size_effect:>11} |"
105        )));
106    }
107    fruits_lines.push(tab_jonction);
108    fruits_lines.push(Line::from(get_button_span(selected)));
109    let nb_lines = fruits_lines.len();
110    //more idiomatic using lines (as in: https://ratatui.rs/recipes/render/display-text/)
111    //can mix style inside the same line using Line::from(vec!["hello".green(),
112    // " ".into(), "world".green().bold(), "3".into()]),
113    frame.render_widget(
114        Paragraph::new(Text::from(fruits_lines)).centered(),
115        frame_vertically_centered_rect(frame.area(), nb_lines),
116    );
117}
118/// Display the speed menu center aligned, vertically centered
119fn speed_menu(frame: &mut Frame, selected: &GreetingOption) {
120    // Speed effects
121    let mut speed_lines = Vec::new();
122    let speed_tab_jonction = Line::from("+------------+------------+----------------+---------+ ");
123    speed_lines.push(speed_tab_jonction.clone());
124    speed_lines.push(Line::from(
125        "| Speed Name | Value (ms) | Score Modifier | Symbol  | ",
126    ));
127    speed_lines.push(speed_tab_jonction.clone());
128    for s in speed::Speed::value_variants() {
129        let SpeedConfig {
130            name,
131            ms_value,
132            score_modifier,
133            symbol,
134        } = s.config();
135        //:<10 and so on are formating options, e.g., saying aligning left with min 10 chars
136        speed_lines.push(Line::from(format!(
137            "| {name:<10} | {ms_value:>10} | {score_modifier:>14} | {symbol:<6} | "
138        )));
139    }
140    speed_lines.push(speed_tab_jonction);
141    speed_lines.push(Line::from(get_button_span(selected)));
142    let nb_lines = speed_lines.len();
143    frame.render_widget(
144        Paragraph::new(Text::from(speed_lines)).centered(),
145        frame_vertically_centered_rect(frame.area(), nb_lines),
146    );
147}

210fn get_button_span(selected: &GreetingOption) -> Vec<Span> {
211    let selected_main: Vec<Span> = vec![" [ ".red(), "Main".into(), " ] ".red().bold()];
212    let default_main: Vec<Span> = vec![" [".into(), "M".yellow(), "ain]".into()];
213    let selected_fruit: Vec<Span> = vec![" [ ".red(), "Fruit".into(), " ] ".red().bold()];
214    let default_fruit: Vec<Span> = vec![" [".into(), "F".yellow(), "ruit]".into()];
215    let selected_velocity: Vec<Span> = vec![" [ ".red(), "Speed".into(), " ] ".red().bold()];
216    let default_velocity: Vec<Span> = vec![" [".into(), "S".yellow(), "peed]".into()];
217    let selected_start: Vec<Span> = vec![" [ ".red(), "Run".into(), " ] ".red().bold()];
218    let default_start: Vec<Span> = vec![" [".into(), "R".yellow(), "un]".into()];
219    let selected_setup: Vec<Span> = vec![" [ ".red(), "Edit⚙️".into(), " ] ".red().bold()];
220    let default_setup: Vec<Span> = vec![" [".into(), "E".yellow(), "dit⚙️]".into()];
221    let selected_help: Vec<Span> = vec![" [ ".red(), "Help".into(), " ] ".red().bold()];
222    let default_help: Vec<Span> = vec![" [".into(), "H".yellow(), "elp]".into()];
223    let mut vec_line_button = vec!["↔".into()];
224    if selected == &GreetingOption::MainMenu {
225        vec_line_button.extend(selected_main);
226    } else {
227        vec_line_button.extend(default_main);
228    }
229    if selected == &GreetingOption::Fruits {
230        vec_line_button.extend(selected_fruit);
231    } else {
232        vec_line_button.extend(default_fruit);
233    }
234    if selected == &GreetingOption::Velocity {
235        vec_line_button.extend(selected_velocity);
236    } else {
237        vec_line_button.extend(default_velocity);
238    }
239    if selected == &GreetingOption::StartGame {
240        vec_line_button.extend(selected_start);
241    } else {
242        vec_line_button.extend(default_start);
243    }
244    if selected == &GreetingOption::Parameters {
245        vec_line_button.extend(selected_setup);
246    } else {
247        vec_line_button.extend(default_setup);
248    }
249    if selected == &GreetingOption::Help {
250        vec_line_button.extend(selected_help);
251    } else {
252        vec_line_button.extend(default_help);
253    }
254    vec_line_button
