140impl GameOptions {
141    /// Returns the initial snake position
142    #[must_use]
143    pub fn initial_position() -> Position {
144        INI_POSITION
145    }
146
147    /// Gets the current speed setting
148    #[must_use]
149    pub fn get_speed(&self) -> Speed {
150        self.speed
151    }
152
153    /// Save the current parameters to a TOML file
154    ///
155    /// # Errors
156    ///
157    /// Returns an error if the file cannot be created or written to.
158    ///
159    /// # Panics
160    ///
161    /// Panics if the game parameters cannot be serialized to TOML.
162    pub fn save_to_toml<P: AsRef<Path>>(&mut self, path: P) -> io::Result<()> {
163        self.save = false;
164        self.load = false;
165        let toml_string =
166            toml::to_string_pretty(self).expect("Failed to serialize GameParameters to TOML");
167        let full_output = format!("{PARAMS_HEADER}\n{toml_string}");
168        let mut file = File::create(path)?;
169        file.write_all(full_output.as_bytes())?;
170        Ok(())
171    }
172
173    /// Load parameters from a TOML file
174    ///
175    /// # Errors
176    ///
177    /// Returns an error if the file cannot be opened or read.
178    ///
179    /// # Panics
180    ///
181    /// Panic if the file contents cannot be deserialized as valid TOML.
182    pub fn load_from_toml<P: AsRef<Path>>(path: P) -> io::Result<Self> {
183        let mut file = File::open(path)?;
184        let mut contents = String::new();
185        file.read_to_string(&mut contents)?;
186        let params =
187            toml::from_str(&contents).expect("Failed to deserialize GameParameters from TOML");
188        Ok(params)
189    }
190    // In real Life, we will validate data more thoroughly
191    pub fn validate_and_adapt(&mut self) {
192        self.nb_of_fruit = self.nb_of_fruit.clamp(1, 9999);
193        self.life = self.life.clamp(1, 9999);
194        self.snake_length = self.snake_length.clamp(1, 999);
195    }
